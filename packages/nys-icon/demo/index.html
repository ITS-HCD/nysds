Code

1. Import and register libraries

import { registerIconLibrary } from '@nysds/nys-icon';

// Font Awesome (free solid icons via jsDelivr CDN)
registerIconLibrary('fa', {
  resolver: name =>
    `https://cdn.jsdelivr.net/npm/@fortawesome/fontawesome-free@6/svgs/solid/${name}.svg`,
  mutator: svg => svg.setAttribute('fill', 'currentColor'),
});

// Google Material Symbols (outlined)
registerIconLibrary('material', {
  resolver: name =>
    `https://fonts.gstatic.com/s/i/short-term/release/materialsymbolsoutlined/${name}/default/24px.svg`,
  mutator: svg => svg.setAttribute('fill', 'currentColor'),
});

2. Use in HTML

<!-- Default NYSDS icons (no library attribute needed) -->
<nys-icon name="check_circle" size="lg" color="#1a7a2e"></nys-icon>

<!-- Font Awesome -->
<nys-icon name="house" library="fa" size="lg"></nys-icon>

<!-- Material Symbols -->
<nys-icon name="home" library="material" size="lg"></nys-icon>
